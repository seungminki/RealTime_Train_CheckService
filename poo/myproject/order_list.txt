# 깃 처음 올릴 때
git init
git add .
git commit -m "message"
git remote add origin "github/your_repo.git"
git push -u origin master

# 깃 업데이트 할 때
git add .
git commit -m "message"
git remote -v
git push -u origin master

만약 master 브랜치가 없어서 발생하는 오류라면:
git checkout -b "master"
git push origin master
